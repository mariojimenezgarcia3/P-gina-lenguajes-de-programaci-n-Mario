── Función que muestra el informe de una zona ── def mostrar_zona(zona): print(f"\n{'═'*40}") print(f" {zona['nombre'].upper()}") print(f"{'═'*40}") if zona["tipo"] == "huerto": print(f" 🌱 Cultivos: {', '.join(zona['cultivos'])}") print(f" 📐 Superficie: {zona['superficie_m2']} m²") print(f" 👥 Responsable: {zona['responsable']}") elif zona["tipo"] == "meteo": d = zona["datos_actuales"] print(f" 🌡️ Temperatura : {d['temp_C']} °C") print(f" 💧 Humedad : {d['humedad_%']} %") print(f" 🌬️ Viento : {d['viento_km_h']} km/h") print(f" 📡 Sensores : {', '.join(zona['sensores'])}") else: practicas = zona.get("practicas", []) capacidad = zona.get("capacidad_alumnos", "—") print(f" 🔧 Prácticas : {', '.join(practicas)}") print(f" 🪑 Capacidad : {capacidad} alumnos") # ── Función principal ── def menu_parque(): print("╔══════════════════════════════════════╗") print("║ PARQUE TECNOLÓGICO — IES TECH ║") print("╚══════════════════════════════════════╝") print("\nZonas disponibles:") for z in zonas_parque: print(f" [{z['id']}] {z['nombre']}") print(" [0] Salir") while True: opcion = input("\nSelecciona una zona: ") if opcion == "0": print("Saliendo del sistema. ¡Hasta pronto!") break zona = next((z for z in zonas_parque if str(z["id"]) == opcion), None) if zona: mostrar_zona(zona) else: print("Opción no válida.") if __name__ == "__main__": menu_parque()
